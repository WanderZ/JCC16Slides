Functions, Methods

						
A.k.a. Subroutines

						
Main Method - the starting point of every JAVA Program

						

public class HelloWorld {
	public HelloWorld() {
		// This is a constructor
	}

	public static void main(String[] args) {
		System.out.println("Hello World");
		sayHello(args);
	}

	public String toString() {
		return "Hello World";
	}

	private void doWantYouWant() {
		// do what i want. Nothing.
		return;
	}

	protected static void sayHello(String[] names) {
		// A program to say hello to everyone.
		for(String person : names) {
			System.out.println("Hallo " + person);
		}
	}

	protected static void sayHello() {
		// A program to say hello to a defined list of people.
		String[] names = {
			"Andrew", 
			"Terence", 
			"Juliana", 
			"Suan Khai"
		};
		for(String person : names) {
			System.out.println("Hallo " + person);
		}
	}
}


					

					

						
Keywords: static, void, etc

						
static is a reserved keyword to load the method or variable into the memory once the program starts - the effect is that you can run it immediately once the program starts. 
(if it starts)
						

						
void is to indicate a "no need to return anything" for the "return type" of the method signature.
						

						
public, private and protected indicates the access type of the method or variable. Optional, but absence of it indicates the default "package" access.
						

						
Here's a whole list of reserved keywords that you cannot use as variables/method names/class names.

public class Box {
	private int length;
	private int breadth;
	private int height;
	public Box() {} 
	public Box(int length, int breadth, int height) {
		this.length 	= length;
		this.breadth 	= breath;
		this.height 	= height;
	}
	public Box(int length) {
		this.length 	= length;
		this.breadth 	= length;
		this.height 	= length;
	}
	public int calculateVolume() {
		return this.length * this.breadth * this.height;
	}
	public int calculateSurfaceArea() {
		return 	2 * this.length * this.breadth 	+
			2 * this.length * this.height 	+
			2 * this.breadth * this.height;	
	}

	public String toString() {
		return 	"Length: " + this.length 		+ 
			"\nBreadth: " + this.breadth 	+
			"\nHeight: " + this.height;
	}
	public boolean equals(Box objBox) {
		return	objBox.length == this.length &&
			objBox.breadth == this.breadth &&
			objBox.height == this.height;
	}
	public int getLength() 	{return this.length;}
	public int getBreadth() {return this.breadth;}
	public int getHeight() 	{return this.height;}
}

					

					

						
OOP: making your own objects 2

						

public class ColouredBox extends Box {
  private String colour;
  public ColouredBox() {super();}
  public ColouredBox(int length, int breadth, int height, String colour) {
    super(length, breadth, height);
    this.colour = colour;
  }
  public ColouredBox(int length, int breadth, int height, String colour) {
    super(length, breadth, height);
    this.colour = colour;
  }
  public ColouredBox(Box box, String colour) {
    super(box.getLength(), box.getBreadth(), box.getHeight());
    this.colour = colour;
  }

  public String toString() {
    return	super.toString() + 
    	"\nColour: " + this.colour;
  }

  public boolean equals(ColouredBox cBox) {
    return 	getLength() == cBox.getLength() &&
    	getHeight() == cBox.getHeight() &&
    	getBreadth() == cBox.getBreadth() &&
    	cBox.colour.equals(this.colour);
  }
}

public class HelloWorld {

    System.out.println(firstBox.calculateSurfaceArea());

    Box secondBox = new Box(1, 2, 3);
    System.out.println(secondBox);
    System.out.println(secondBox.calculateVolume());
    System.out.println(secondBox.calculateSurfaceArea());

      "'=='? " + (firstBox == secondBox));

      "equals()? " + firstBox.equals(secondBox));

      "'=='?" + 
      rainbowBox == rainbowCube ? "Y" : "N");
    System.out.println(
      "equals()?" +

}

import java.util.Scanner;

public class HelloWorld {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    // <Datatype> <Your Variable Name> = "new" <Constructor>(<Parameters>);

    System.out.println("Max of your two numbers is: " + max);

    System.out.print("The middle letter of this string is: ");
